244  {
245  memset(this->data, 0, sizeof(this->data));
246  double rad = DEG_TO_RAD(angle);
247  float c = (float)cos(rad);
248  float s = (float)sin(rad);
249  Vector3<float> v = axis.normalized();
250  float c1 = 1.0f - c;
251  float xyc1 = v.x * v.y * c1;
252  float yzc1 = v.y * v.z * c1;

259  this->data[8] = zxc1 + ys; this->data[9] = yzc1 - xs; this->data[10] = v.z * v.z * c1 + c;
