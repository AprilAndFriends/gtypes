249  inline void setRotation(const Vector3& axis, float angle)
250  {
251  memset(this->data, 0, sizeof(this->data));
252  double rad = DEG_TO_RAD(angle);
253  float c = (float)cos(rad);
254  float s = (float)sin(rad);
255  Vector3 v = axis.normalized();
256  float c1 = 1.0f - c;
257  float xyc1 = v.x * v.y * c1;
258  float yzc1 = v.y * v.z * c1;
259  float zxc1 = v.z * v.x * c1;
260  float xs = v.x * s;
261  float ys = v.y * s;
262  float zs = v.z * s;
263  this->data[0] = v.x * v.x * c1 + c; this->data[1] = xyc1 + zs; this->data[2] = zxc1 - ys;
264  this->data[4] = xyc1 - zs; this->data[5] = v.y * v.y * c1 + c; this->data[6] = yzc1 + xs;
265  this->data[8] = zxc1 + ys; this->data[9] = yzc1 - xs; this->data[10] = v.z * v.z * c1 + c;
266  this->data[15] = 1.0f;
267  }
